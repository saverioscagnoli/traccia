1use crate::{LogLevel, target::TargetId};
2use std::sync::{LazyLock, RwLock};
3
4type HookCallback = Box<dyn Fn(LogLevel, &TargetId) + Send + Sync>;
5
6/// Represents a hook that can be set to trigger
7/// at specific points in the logging process.
8///
9/// Hooks can be used to perform custom actions, for example,
10/// for performing side effects, such as clearing the console
11/// or sending notifications.
12///
13/// Hooks are executed in the order they are added.
14pub enum Hook {
15    /// Hook that is called before a log message is written.
16    BeforeLog(HookCallback),
17    /// Hook that is called after a log message is written.
18    /// This is useful for post-processing or additional actions.
19    AfterLog(HookCallback),
20}
21
22pub struct HookSystem {
23    before_log_hooks: Vec<Hook>,
24    after_log_hooks: Vec<Hook>,
25}
26
27impl HookSystem {
28    pub fn new() -> Self {
29        Self {
30            before_log_hooks: Vec::new(),
31            after_log_hooks: Vec::new(),
32        }
33    }
34
35    /// Splits the hooks into two separate vectors
36    pub fn add_hook(&mut self, hook: Hook) {
37        match hook {
38            Hook::BeforeLog(_) => self.before_log_hooks.push(hook),
39            Hook::AfterLog(_) => self.after_log_hooks.push(hook),
40        }
41    }
42
43    /// Triggers all before log hooks
44    /// with the provided log level and target ID.
45    pub fn trigger_before_log(&self, level: LogLevel, target_id: &TargetId) {
46        for hook in &self.before_log_hooks {
47            if let Hook::BeforeLog(callback) = hook {
48                callback(level, target_id);
49            }
50        }
51    }
52
53    /// Triggers all after log hooks
54    /// with the provided log level and target ID.
55    pub fn trigger_after_log(&self, level: LogLevel, target_id: &TargetId) {
56        for hook in &self.after_log_hooks {
57            if let Hook::AfterLog(callback) = hook {
58                callback(level, target_id);
59            }
60        }
61    }
62}
63
64static HOOK_SYSTEM: LazyLock<RwLock<HookSystem>> = LazyLock::new(|| RwLock::new(HookSystem::new()));
65
66pub fn hook_system() -> &'static RwLock<HookSystem> {
67    &HOOK_SYSTEM
68}
69
70/// Sets a hook to be called at the specified point in the logging process.
71///
72/// Multiple hooks can be set.
73///
74/// IMPORTANT: do not use the macros defined in this crate in `BeforeLog` or `AfterLog` hooks.
75/// This will lead to an infinite loop, as the logging macros trigger the hooks
76/// themselves.
77pub fn set_hook(hook: Hook) {
78    if let Ok(mut hook_system) = HOOK_SYSTEM.write() {
79        hook_system.add_hook(hook);
80    } else {
81        eprintln!("Failed to acquire write lock on hook system. Hook not set.");
82    }
83}
