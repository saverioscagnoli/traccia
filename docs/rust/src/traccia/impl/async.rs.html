1use crate::{Config, DefaultFormatter, Formatter, LogLevel, Logger, Record, Target, hooks};
2use std::{
3    sync::{Mutex, mpsc},
4    thread,
5};
6
7enum ChannelMessage {
8    Log(String, LogLevel),
9    Flush,
10}
11
12pub struct DefaultLogger {
13    config: Config,
14    sender: mpsc::Sender<ChannelMessage>,
15    worker: Mutex<Option<thread::JoinHandle<()>>>,
16}
17
18impl DefaultLogger {
19    pub fn new(config: Config) -> Self {
20        let (sender, receiver) = mpsc::channel();
21
22        let thread_targerts = config.targets.clone();
23        let worker = std::thread::spawn(move || {
24            Self::worker_thread(receiver, thread_targerts);
25        });
26
27        DefaultLogger {
28            config,
29            sender,
30            worker: Mutex::new(Some(worker)),
31        }
32    }
33
34    fn process_message(formatted: &str, level: LogLevel, targets: &[Box<dyn Target>]) {
35        // Acquire the hook system lock
36        // This is a read lock, so it won't block other threads from reading
37        // but will block if another thread is writing
38        // So, it fails only if the user tries to set a hook while the logger is running,
39        // which is not encouraged.
40        let hook_system = hooks::hook_system().read().expect(
41            "Failed to acquire the hook system lock. You should use `set_hook` before initializing the logger.",
42        );
43
44        for target in targets {
45            // Check if the target has a custom filter level
46            if let Some(filter_level) = target.filter_level() {
47                if level < filter_level {
48                    continue;
49                }
50            }
51
52            let target_id = target.id();
53
54            hook_system.trigger_before_log(level, &target_id);
55
56            if let Err(e) = target.write(level, &formatted) {
57                eprintln!("Failed to write to target: {}", e);
58            }
59
60            hook_system.trigger_after_log(level, &target_id);
61        }
62    }
63
64    fn worker_thread(receiver: mpsc::Receiver<ChannelMessage>, targets: Vec<Box<dyn Target>>) {
65        loop {
66            match receiver.recv() {
67                Ok(ChannelMessage::Log(formatted, level)) => {
68                    Self::process_message(&formatted, level, &targets)
69                }
70
71                Ok(ChannelMessage::Flush) => break,
72
73                Err(_) => break,
74            }
75        }
76
77        // Drain the remaining messages
78        while let Ok(message) = receiver.try_recv() {
79            match message {
80                ChannelMessage::Log(formatted, level) => {
81                    Self::process_message(&formatted, level, &targets)
82                }
83
84                _ => {}
85            }
86        }
87    }
88}
89
90impl Logger for DefaultLogger {
91    fn enabled(&self, level: crate::LogLevel) -> bool {
92        self.config.level <= level
93    }
94
95    fn abort(&self) {
96        let _ = self.sender.send(ChannelMessage::Flush);
97        if let Ok(mut handle) = self.worker.lock() {
98            if let Some(handle) = handle.take() {
99                handle.join().unwrap();
100            }
101        } else {
102            eprintln!("Cleanup process failed. Some final logs may not be written.");
103        }
104    }
105
106    fn log(&self, record: &Record) {
107        if !self.enabled(record.level) {
108            return;
109        }
110
111        let formatted = match &self.config.format {
112            Some(formatter) => formatter.format(record),
113            None => DefaultFormatter.format(record),
114        };
115
116        let _ = self
117            .sender
118            .send(ChannelMessage::Log(formatted, record.level));
119    }
120}
121
122impl Default for DefaultLogger {
123    fn default() -> Self {
124        DefaultLogger::new(Config::default())
125    }
126}
