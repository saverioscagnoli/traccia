5/// Provides utilities for adding color to log messages in terminal output.
6/// These colors are automatically stripped when writing to non-terminal

9/// Terminal colors for text output.
10///
11/// Represents the standard 8 ANSI colors available in most terminals.
12#[derive(Debug, Clone, Copy, PartialEq, Eq)]
13pub enum Color {

15    Black,
16    /// Red color (ANSI code 31 - bg: 41)
17    Red,
18    /// Green color (ANSI code 32 - bg: 42)
19    Green,
20    /// Yellow color (ANSI code 33 - bg: 43)
21    Yellow,
22    /// Blue color (ANSI code 34 - bg: 44)
23    Blue,
24    /// Magenta color (ANSI code 35 - bg: 45)
25    Magenta,
26    /// Cyan color (ANSI code 36 - bg: 46)
27    Cyan,
28    /// White color (ANSI code 37 - bg: 47)
29    White,
30
31    /// Default color (ANSI code 39)
32    Default,
33
34    /// Bright black color (ANSI code 90 - bg: 100)
35    BrightBlack,
36
37    /// Bright red color (ANSI code 91 - bg: 101)
38    BrightRed,
39
40    /// Bright green color (ANSI code 92 - bg: 102)
41    BrightGreen,
42
43    /// Bright yellow color (ANSI code 93 - bg: 103)
44    BrightYellow,
45
46    /// Bright blue color (ANSI code 94 - bg: 104)
47    BrightBlue,
48
49    /// Bright magenta color (ANSI code 95 - bg: 105)
50    BrightMagenta,

52    /// Bright cyan color (ANSI code 96 - bg: 106)
53    BrightCyan,
54
55    /// Bright white color (ANSI code 97 - bg: 107)
56    BrightWhite,

65impl Color {
66    /// Returns the ANSI escape code for the color.
67    ///
68    /// # Returns
69    ///
70    /// The ANSI escape sequence string for the color
71    pub fn ansi_code_foreground(&self) -> String {
72        match self {
73            Color::Black => "\x1b[30m".to_string(),
74            Color::Red => "\x1b[31m".to_string(),
75            Color::Green => "\x1b[32m".to_string(),
76            Color::Yellow => "\x1b[33m".to_string(),
77            Color::Blue => "\x1b[34m".to_string(),
78            Color::Magenta => "\x1b[35m".to_string(),
79            Color::Cyan => "\x1b[36m".to_string(),
80            Color::White => "\x1b[37m".to_string(),
81            Color::Default => "\x1b[39m".to_string(),
82            Color::BrightBlack => "\x1b[90m".to_string(),
83            Color::BrightRed => "\x1b[91m".to_string(),
84            Color::BrightGreen => "\x1b[92m".to_string(),
85            Color::BrightYellow => "\x1b[93m".to_string(),
86            Color::BrightBlue => "\x1b[94m".to_string(),
87            Color::BrightMagenta => "\x1b[95m".to_string(),
88            Color::BrightCyan => "\x1b[96m".to_string(),
89            Color::BrightWhite => "\x1b[97m".to_string(),
90            Color::ID(id) => format!("\x1b[38;5;{}m", id),
91            Color::RGB(r, g, b) => format!("\x1b[38;2;{};{};{}m", r, g, b),
92        }
93    }

120/// Trait for applying colors to strings.
121///
122/// This trait provides the ability to color text using ANSI escape codes.
123pub trait Colorize: Display {
124    /// Applies a color to the string.
125    ///
126    /// # Arguments
127    ///
128    /// * `color` - The color to apply
129    ///
130    /// # Returns
131    ///
132    /// A new string with the color applied via ANSI escape codes
133    fn color(&self, color: Color) -> String {
134        format!("{}{}\x1b[0m", color.ansi_code_foreground(), self)
135    }
136
137    /// Applies a background color to the string.
138    ///
139    /// # Arguments
140    ///
141    /// * `color` - The color to apply
142    ///
143    /// # Returns
144    ///
145    /// A new string with the background color applied via ANSI escape codes
146    fn background(&self, color: Color) -> String {
147        format!("{}{}\x1b[0m", color.ansi_code_background(), self)
148    }
149}

151impl Colorize for str {}
152impl Colorize for String {}

154pub trait Style: Display {
155    /// Applies bold style to the string.
156    fn bold(&self) -> String {
157        format!("\x1b[1m{}\x1b[0m", self)
158    }
159
160    /// Dims the color of the string.
161    fn dim(&self) -> String {
162        format!("\x1b[2m{}\x1b[0m", self)
163    }
164
165    /// Applies italic style to the string.
166    fn italic(&self) -> String {
167        format!("\x1b[3m{}\x1b[0m", self)
168    }
169
170    /// Underlines the string.
171    fn underline(&self) -> String {
172        format!("\x1b[4m{}\x1b[0m", self)
173    }
174
175    /// Makes the content of the string blink.
176    fn blink(&self) -> String {
177        format!("\x1b[5m{}\x1b[0m", self)
178    }
179
180    /// Reverses the foreground and background colors of the string.
181    fn reverse(&self) -> String {
182        format!("\x1b[7m{}\x1b[0m", self)
183    }
184
185    /// Hides the content of the string with the background color.
186    fn hidden(&self) -> String {
187        format!("\x1b[8m{}\x1b[0m", self)
188    }
189
190    /// Strikes through the content of the string.
191    fn striketrough(&self) -> String {
192        format!("\x1b[9m{}\x1b[0m", self)
193    }
194}
